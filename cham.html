const SHEET_NAME = "Results_v7"; function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sh = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME); if (sh.getLastRow() === 0) { sh.appendRow([ "id", "timestamp", "firstChoice", "topicChoice", "route", "screenout", "screenoutReason", "sqGender", "sqBranch", "sqRank", "sqPortalNews", "sqMobilePhone", "sqInternetTime", "sqPolitical", "sqDevice", "fact1", "fact2", "fact3", "fact4", "fact5", "algoQ7", "algoQ8", "algoQ9", "algoQ10", "algoQ11", "algoQ12", "selectedNews", "selectedNewsTitle", "selectedNewsPage", "q2", "q3", "userAgent" ]); } const body = (e && e.postData && typeof e.postData.contents === "string") ? e.postData.contents : "{}"; let data = {}; try { data = JSON.parse(body || "{}"); } catch (err) { data = {}; } const id = data.id || Utilities.getUuid(); const row = [ id, data.timestamp || new Date().toISOString(), data.firstChoice || "", data.topicChoice || "", data.route || "", data.screenout || "N", data.screenoutReason || "", data.sqGender || "", data.sqBranch || "", data.sqRank || "", data.sqPortalNews || "", data.sqMobilePhone || "", data.sqInternetTime || "", data.sqPolitical || "", data.sqDevice || "", data.fact1 || "", data.fact2 || "", data.fact3 || "", data.fact4 || "", data.fact5 || "", data.algoQ7 || "", data.algoQ8 || "", data.algoQ9 || "", data.algoQ10 || "", data.algoQ11 || "", data.algoQ12 || "", data.selectedNews || "", data.selectedNewsTitle || "", data.selectedNewsPage || "", data.q2 || "", data.q3 || "", data.userAgent || "" ]; sh.appendRow(row); return ContentService .createTextOutput(JSON.stringify({ ok: true, id: id })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ ok: false, error: String(err) })) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sh = ss.getSheetByName(SHEET_NAME); const action = ((e && e.parameter && e.parameter.action) || "").toLowerCase(); const limit = Math.min(parseInt((e && e.parameter && e.parameter.limit) || "200", 10) || 200, 2000); let payload = { ok: true, rows: [] }; if (action === "list" && sh && sh.getLastRow() >= 2) { const values = sh.getDataRange().getValues(); const header = values[0]; const dataRows = values.slice(1).reverse().slice(0, limit); payload.rows = dataRows.map(r => { const obj = {}; header.forEach((k, i) => obj[k] = r[i]); return obj; }); } const cb = e && e.parameter ? e.parameter.callback : ""; if (cb) { const js = `${cb}(${JSON.stringify(payload)})`; return ContentService .createTextOutput(js) .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(JSON.stringify(payload)) .setMimeType(ContentService.MimeType.JSON); } catch (err) { const payload = { ok: false, error: String(err), rows: [] }; const cb = e && e.parameter ? e.parameter.callback : ""; if (cb) { const js = `${cb}(${JSON.stringify(payload)})`; return ContentService .createTextOutput(js) .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(JSON.stringify(payload)) .setMimeType(ContentService.MimeType.JSON); } }
